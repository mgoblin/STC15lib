#include <delay.h>
#define LED P10
void main()
{

{
// set system clock freq divider value to master clock to slowdown MCU
set_frequency_divider_scale(divider);
LED = !LED;
// but delay still 500 ms because delay know about frequency divider
delay_ms(500);
}
}
