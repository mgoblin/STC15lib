133#define adc_read(value) \
134{ \
135 /* Clear ADC result ready flag */ \
136 bit_clr(ADC_CONTR, ~(1 << ADC_FLAG_BIT)); \
137 /* Set ADC power on */ \
138 bit_set(ADC_CONTR, 1 << ADC_START_BIT); \
